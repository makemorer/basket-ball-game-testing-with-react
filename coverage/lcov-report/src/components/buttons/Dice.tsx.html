import React, {useContext} from 'react'
import dice5 from '../../dice-5.png'
import dice1 from '../../dice-1.png'
import dice2 from '../../dice-2.png'
import dice3 from '../../dice-3.png'
import dice4 from '../../dice-4.png'
import dice6 from '../../dice-6.png'
import { DiceContext } from './Button'
 
 
export const Dice = () => {
  const dice = useContext(DiceContext)
  if(dice === 1){
    var img
    img = <img src={dice1} alt="Dice" className="dice"/>
  }else if(dice === 2){
    img = <img src={dice2} alt="Dice" className="dice"/>
  }else if(dice === 3){
    img = <img src={dice3} alt="Dice" className="dice"/>
  }else if(dice === 4){
    img = <img src={dice4} alt="Dice" className="dice"/>
  }else if(dice === 5){
    img = <img src={dice5} alt="Dice" className="dice"/>
  }else if(dice === 6){
    img = <img src={dice6} alt="Dice" className="dice"/>
  }
    return(
      <React.Fragment>{img}</React.Fragment>
    )
  
}
